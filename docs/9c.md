<template>
  <div>
    <div class="gc">
      <div class="td" v-for="(e,i) in y.split(',')" :key="'y'+i">{{ e }}</div>
    </div>
    <hr>
    <div class="gc">
      <div class="td" v-for="(e,i) in x" :key="'x'+i">{{ e }}</div>
    </div>
  </div>
</template>

<script>
export default {
  data() {
    return {
      x: Array(90)
        .fill(0)
        .map((_, i) => i + 1)
        .sort(() => Math.random() - 1 / 2)
        .slice(0, 5)
        .sort((a, b) => a - b),
      y: localStorage.myArray
    };
  },
  mounted() {
    localStorage.myArray = this.x;
  }
};
</script>
